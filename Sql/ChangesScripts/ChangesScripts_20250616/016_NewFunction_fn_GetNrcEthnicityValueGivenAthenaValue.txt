/* -----------------------------------------------------------------------------------------------------------
   Function Name  :  nrc.fn_GetNrcEthnicityValueGivenAthenaValue
   Business Analyis:
   Project/Process :   
   Description     :  If the inputted Athena Ethnicity Value is either blank or does not exist in the EthnicityCrossWalk table,
                      output Error Message along with the default NRC Value.  The calling procedure will parse the default value
					  from this message.  If the inputted Athena Ethnicity does exist in that table, output the corresponding 
					  NrcValue.
                        
   Author          :   Philip Morrison
   Create Date     :   6/13/2025

-- create function scalar-valued udf
CREATE FUNCTION [nrc].[fn_GetNrcEthnicityValueGivenAthenaValue]
(
  @inputAthenaEthnicityValue [varchar] (100)
) 
RETURNS [nvarchar] (100)
AS
BEGIN
   
DECLARE @returnString [nvarchar](100) = '';

-- If the inputted Ethnicity value is blanks, merely put out the default value.

if (LEN(ltrim(rtrim(@inputAthenaEthnicityValue))) = 0) BEGIN
   SET @returnString = 'U';
   RETURN @returnString;
END

DECLARE @existsAthenaValue [varchar] (100) = null;
DECLARE @existsNrcValue [varchar] (100) = null;

-- Look up the inputted Athena value in the EthnicityCrossWalk table.

SELECT @existsAthenaValue = [AthenaValue]
       ,@existsNrcValue = [NrcValue]
FROM [Nrc].[nrc].[EthnicityCrossWalk]
WHERE [AthenaValue] = @inputAthenaEthnicityValue;

-- If the inputted Athena value does not exist, put out the error message with the default value.

IF (@existsAthenaValue IS NULL) BEGIN
   SET @returnString = 'U';
   RETURN @returnString;
END

-- If the inputted Athena value does exist in the table, put out the error message with the default value.

SET @returnString = @existsNrcValue;

RETURN @returnString;

END
